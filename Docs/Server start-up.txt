Letsencrypt не работает на IP без доменного имени, поэтому ставим mkcert (https://github.com/FiloSottile/mkcert)

sudo apt install libnss3-tools

curl -JLO "https://dl.filippo.io/mkcert/latest?for=linux/amd64"
chmod +x mkcert-v*-linux-amd64
sudo cp mkcert-v*-linux-amd64 /usr/local/bin/mkcert

mkcert -install
mkcert 0.0.0.0
uvicorn --ssl-keyfile=0.0.0.0-key.pem --ssl-certfile=0.0.0.0.pem api:app --host 0.0.0.0 --port 8000
